Production-Ready System Specification: The Gauntlet Engine
This document provides a cohesive, unified, and noise-free architectural spec designed directly for Claude Code to implement the core product backend, frontend, AI orchestration layer, database, and ElevenLabs API integration.

The platform consists of a single-page application (SPA) front-end, a robust REST API backend, and an intelligent AI Orchestration layer that drives multi-lens evaluation pipelines without hardcoded logic.

2. Database Schema
The Postgres database structure maps the entire assessment lifecycle, ensuring full audit compliance, idempotent evaluation steps, and state preservation for judge swapping or evidence updates.
SQL
-- Core User Account Data
CREATE TABLE users (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
email VARCHAR(255) UNIQUE NOT NULL,
role VARCHAR(50) NOT NULL DEFAULT 'user',
org_id UUID,
preferences JSONB DEFAULT '{}'::jsonb,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- User Idea Submissions
CREATE TABLE submissions (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
user_id UUID REFERENCES users(id) ON DELETE CASCADE,
type VARCHAR(50) NOT NULL, -- 'idea', 'product', 'claim', 'story', 'prototype', 'health_concept', 'research_concept'
title VARCHAR(255) NOT NULL,
description TEXT NOT NULL,
goal_audience TEXT,
environment TEXT,
constraints TEXT,
self_assessment JSONB,
status VARCHAR(50) NOT NULL DEFAULT 'draft', -- 'draft', 'in_evaluation', 'complete'
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Parsed Structural Elements of a Submission
CREATE TABLE claims (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
submission_id UUID REFERENCES submissions(id) ON DELETE CASCADE,
text TEXT NOT NULL,
type VARCHAR(50),
strength VARCHAR(50),
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Supporting Documentation or References
CREATE TABLE evidence_items (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
submission_id UUID REFERENCES submissions(id) ON DELETE CASCADE,
claim_id UUID REFERENCES claims(id) ON DELETE SET NULL,
source TEXT NOT NULL,
type VARCHAR(50), -- 'RCT', 'observational', 'citation_text', 'url', 'file_upload'
relevance VARCHAR(50), -- 'direct', 'indirect', 'weak'
quality VARCHAR(50), -- 'high', 'medium', 'low'
notes TEXT,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Master Evaluation Record
CREATE TABLE evaluation_runs (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
submission_id UUID REFERENCES submissions(id) ON DELETE CASCADE,
triad UUID[] NOT NULL, -- Array of 3 Judge IDs
status VARCHAR(50) NOT NULL DEFAULT 'pending', -- 'pending', 'processing', 'conflict_resolution', 'complete'
model_version VARCHAR(50) NOT NULL,
orchestration_version VARCHAR(50) NOT NULL,
risk_score NUMERIC(3, 2),
risk_tier VARCHAR(50),
composite_score NUMERIC(3, 2),
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
completed_at TIMESTAMP WITH TIME ZONE
);
-- Individual Judge Analysis Outputs
CREATE TABLE judge_evaluations (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
judge_id VARCHAR(50) NOT NULL,
stage VARCHAR(50) NOT NULL, -- 'clarity', 'evidence', 'risk', 'feasibility', 'impact', 'final'
dimension_scores JSONB NOT NULL, -- Vector tracking: structure, viability, risk, narrative, evidence, clinical, cultural, psych
raw_output TEXT NOT NULL,
confidence NUMERIC(3, 2) NOT NULL,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Cross-Analysis Mapping Matrix
CREATE TABLE triangulations (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
triangulation_matrix JSONB NOT NULL, -- Agreement, conflict, and coverage tracking
heatmap JSONB NOT NULL,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Risk Architecture Drivers
CREATE TABLE risk_assessments (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
risk_tier VARCHAR(50) NOT NULL,
risk_score NUMERIC(3, 2) NOT NULL,
risk_drivers JSONB NOT NULL, -- dimensionRisk, riskFlags, subDimensionSeverity, conflictPenalty
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- Tracking Modality Changes via Action Items
CREATE TABLE evidence_assessments (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
evidence_status VARCHAR(50) NOT NULL,
evidence_gaps JSONB NOT NULL,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
-- ElevenLabs Synthesized Files
CREATE TABLE audio_clips (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
judge_id VARCHAR(50) NOT NULL,
type VARCHAR(50) NOT NULL, -- 'micro_comment', 'stage_summary', 'final_summary', 'conflict_explanation'
storage_url TEXT NOT NULL,
duration INT NOT NULL,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE reports (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
evaluation_run_id UUID REFERENCES evaluation_runs(id) ON DELETE CASCADE,
report_object JSONB NOT NULL,
pdf_url TEXT,
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

5. Core Algorithmic Frameworks
A. Judge Selection Recommendation Engine
When a user finishes intake, the system extracts claims and generates a required requirement vector:
$$R = [structure, viability, risk, narrative, evidence, clinical, cultural, psych]$$
Each dimension is explicitly valued between 0 and 1.
The engine computes the cosine similarity between $R$ and each Judge's built-in vector signature ($Lj$).
Valid combinations of 3 judges are generated adhering to constraints: No duplicate lenses, no overlapping blind spots.
Every potential triad is scored via the following optimizing equation:
$$\text{triadScore} = 0.35 \cdot \text{COV} + 0.25 \cdot \text{COMP} + 0.20 \cdot \text{TYPE} + 0.10 \cdot \text{RISK} + 0.10 \cdot \text{EVID}$$
Coverage ($\text{COV}$): Percentage of active dimensions in $R$ addressed by the combination.
Complementarity ($\text{COMP}$): Distance matrix tracking divergence between individual judge competencies.
Type Weighting ($\text{TYPE}$): Submission contextual matching weight (e.g., Clinical concepts boost judge_clinical and judge_evidence).
Risk/Evidence Adequacy ($\text{RISK}$, $\text{EVID}$): Flags that boost analytical sensitivity scores when risk or validation thresholds are high.
B. The 3-Judge Triangulation Mechanism
During active pipelines, judges output isolated 5-dimensional vectors containing normalized scores from 0 to 1.
[Dimension Evaluation Loop]
/      |      \
/       |       \
Judge A Vector  Judge B Vector  Judge C Vector
\       |       /
\      |      /
[Dimension-by-Dimension Comparison]
/               |               \
Spread &lt;= 0.15          Spread &gt;= 0.35         Uncovered
v                       v                    v
[Agreement Zone]       [Conflict Zone]       [Coverage Gap]
Consensus Finding     Blindspot Exposure     Warning Generated
Agreement Zone: Identified where all three chosen judges return values inside a $\pm 0.15$ spread. This explicitly establishes high-confidence consensus.
Conflict Zone: Generated when the mathematical spread across active outputs is $\ge 0.35$. This exposes critical system blindspots.
Weighted Composite Scoring:
$$S = \sum (\text{judge\_weight} \cdot \text{judge\_vector})$$
C. Risk Tier &amp; Assessment Calculation
The absolute Risk Tier (Low, Moderate, High, Critical) is determined by an aggregated risk engine scoring framework:

dimensionRisk: Baseline mathematical mean across risk vectors.
riskFlags: Scalar increases applied per structural exception thrown by judge_ethics or judge_clinical.
conflictPenalty: Weighted additions applied when conflict spreads on safety dimensions exceed $\ge 0.35$.
6. Dynamic Evaluation &amp; Mutation Workflows
A. The 5-Stage Pipeline Heuristics
Submissions cycle through five sequential passes. Judges review each pass strictly filtered through their functional analytical lens:
Stage 1: Clarity
judge_structural -&gt; Checks structural layout and internal logic consistency.
judge_narrative -&gt; Checks story structure and readability alignment.
judge_psych -&gt; Measures entry barrier overhead and user friction.
judge_evidence -&gt; Audits mechanistic definitions and claims layout.
Stage 2: Evidence
judge_evidence -&gt; Scores data strength and reference validity.
judge_clinical -&gt; Verifies scientific plausibility and data parameters.
judge_cultural -&gt; Inspects situational field data and case records.
judge_structural -&gt; Audits internal validation logic for fallacies.
Stage 3: Risk
judge_ethics -&gt; Scans for systemic vulnerabilities and downstream misuse.
judge_clinical -&gt; Analyzes biological or operational vulnerabilities.
judge_cultural -&gt; Flags socio-environmental complications.
judge_viability -&gt; Flags single points of operational failure.
Stage 4: Feasibility
judge_viability -&gt; Gauges real-world execution, overhead, and constraints.
judge_structural -&gt; Evaluates implementation logic dependencies.
Stage 5: Impact
judge_psych -&gt; Projects behavioral changes and retention metrics.
judge_narrative -&gt; Measures market messaging resonance.
judge_ethics -&gt; Weights public-good outputs.
B. State-Preserving Judge Swap Workflow
When a judge is replaced via the UI mid-run, the engine preserves historical data, avoiding unnecessary API or compute execution:
[User Triggers Swap]
v
1. Freeze Existing Data -------------------&gt; Keeps untouched evaluations,
v                                     triangulation rows, &amp; heatmaps.
2. Purge Target Judge Records -------------&gt; Drops old judge outputs from database.
v
3. Execute New Judge Pipeline -------------&gt; Runs 5-stage evaluation loop for
v                                     newly added judge only.
4. Matrix Recalculation -------------------&gt; Recomputes Triangulation Matrix &amp;
v                                     Risk Tier with new composite vector.
5. Update Complete Evaluation Object ------&gt; Returns mutated report payload.
C. Evidence Augmentation Loop
Triggers automatically if the evidence score falls $&lt; 0.45$, or if judge_evidence/judge_clinical throws an explicit validation error.
[System Identifies Gap] -&gt; [Isolate to Claim ID] -&gt; [Render Targeted UI Prompt]
|
[Recalculate Matrix &amp; Risk] &lt;- [Re-Run Evidential Lenses] &lt;- [User Uploads Source]
Isolate Gaps: Gaps are mapped to a distinct JSON object:
JSON
{ "claimId": "CLM-002", "gapType": "Data sufficiency", "severity": "High" }
Targeted Prompts: The interface prompts the user for specific evidence (e.g., "For Claim 2, do you have any human trial data or peer-reviewed citations supporting this effect?").
Targeted Re-Evaluation: On submission, only the affected dimensions for active judges are rescored. The rest of the engine state remains frozen.

9. Frontend User Experience Layout Architecture
The user interface is engineered to visualize multi-lens structural analysis without unnecessary UI clutter.
A. Interactive Submission Dashboard
Displays a paginated master grid tracking all active user entries.
Each row features a clean metadata readout: Submission Title, Type, Status Indicator, Calculated Risk Tier Badge, and Composite Score Trend Line.
Primary Call to Action: New Submission Intake Engine.
B. Triangulation Matrix &amp; Heatmap Grid
Implements an interactive data table displaying the calculated outputs of the three active judges across evaluated dimensions.
Rows indicate system dimensions; columns track individual judge scoring outputs.
The Heatmap Channel: Cells dynamically apply custom status styling depending on the values:
Agreement Zones (Low Delta) render in clean, muted tones.
Conflict Zones ($\ge 0.35$ Variance) flash clear alert flags alongside a prominent Drill Deeper button.
Coverage Gaps (Null Values) show high-contrast missing indicators with an integrated Swap Judge recommendation utility.
C. Active Pipeline Matrix View
A multi-column kanban execution panel rendering live progression through the 5 evaluation passes (Clarity $\rightarrow$ Evidence $\rightarrow$ Risk $\rightarrow$ Feasibility $\rightarrow$ Impact).
As each judge process completes, its specific grid cell fills with the calculated score, actionable micro-critiques, and a localized inline audio player node wired to the ElevenLabs synthesis stream.

5. Algorithmic Mathematical Formulations
A. Judge Selection Matrix Recommendation
When a user finishes input ingestion, the platform maps the submission contents into a target requirement vector:
R=[structure,viability,risk,narrative,evidence,clinical,cultural,psych]
The engine calculates the cosine similarity match score against each judge's primary lens profile (Lj):
matchj​=∥R∥∥Lj​∥R⋅Lj​​
All available judges are combined into arrays of exactly 3 candidates. Triads containing duplicate lenses or matching blind spots are flagged as invalid.
Every functional triad is scored and sorted according to the optimization equation:
triadScore=0.35⋅COV+0.25⋅COMP+0.20⋅TYPE+0.10⋅RISK+0.10⋅EVID
Coverage (COV): The sum of active elements inside R explicitly covered by the candidate triad.
Complementarity (COMP): The spatial distance tracking divergence among chosen competency vectors, preventing analytical groupthink.
Type Weighting (TYPE): Contextual modifier derived from submission classifications (e.g., a Health Concept scales up base weights for judge_doctor and judge_nutritionist).
Risk/Evidence Sensitivity multipliers (RISK, EVID): Scalers that amplify analytical precision when raw inputs contain validation deficiencies or extreme liability signals.
B. The 3-Judge Dimension Triangulation Pipeline
During pipeline execution, each judge outputs an isolated 5-dimensional evaluation vector tracking normalized floats from 0 to 1. The engine executes a cell-by-cell matrix cross-comparison:
Agreement Zone: Calculated where the value delta between all three judges across an active dimension sits within a tight bound of ±0.15. This defines high-confidence system consensus.
Conflict Zone: Triggered when the mathematical spread between the highest and lowest scores across a single dimension matches or exceeds a variance threshold of ≥0.35. This automatically flags system blind spot exposures.
Dynamic Triad Weighting Composite:
S=j=1∑3​(finalWeightj​⋅judge_vectorj​)
finalWeightj​=∑(finalWeights)baseWeightj​×relevanceFactorj​×userSelectionFactorj​​
C. Risk Tier Aggregation Matrix
The system outputs an explicit risk classification (Low, Moderate, High, Critical) computed through an additive penalty scoring framework:
Rfinal​=min(1.0,D+F+S+C+W)
Dimension Risk (D): The baseline weighted mean computed across all threat-facing vectors: Risk, Clinical, Cultural, and Psychological.
Risk Flag Multiplier (F): A severity scaler. Add +0.15 for each unique instance where an individual judge returns a value &gt;0.7 inside a threat dimension, and an additional +0.05 if multiple judges flag the identical dimension.
Sub-Dimension Severity (S): The sum of the top 3 highest calculated sub-dimension risk ratings (Low: 0.05, Moderate: 0.10, High: 0.20, Critical: 0.30).
Conflict Penalty (C): Captures analytical uncertainty as structural risk. Where a variance or split exists on a key safety dimension, penalties are injected into the model calculations:
Mild Conflict (0.35 to 0.45 Spread): +0.05
Moderate Conflict (0.45 to 0.55 Spread): +0.10
Severe Conflict (≥0.55 Spread): +0.20
Contextual Type Modifier (W): Fixed regulatory risk baseline weights adjusted by submission category parameters: Health Concept: +0.20; Research Concept: +0.15; Claim: +0.15; Product/Prototype: +0.10; Idea: +0.05; Story: +0.03.

package pipeline
import (
"context"
"io"
"net/http"
"strings"
"sync"
)
// DispatchAudioGeneration handles non-blocking chunked voice synthesis
func DispatchAudioGeneration(ctx context.Context, runID string, judgeID string, fullScript string) {
// Step 1: Split script by sentence structures to keep ElevenLabs payload small and swift
chunks := SplitScriptIntoSentences(fullScript, 2) // Max 2 sentences per API request
var wg sync.WaitGroup
for idx, chunk := range chunks {
wg.Add(1)
go func(index int, textChunk string) {
defer wg.Done()
// Call the client execution module (as mapped in the primary specification)
resp, err := CallElevenLabsAPI(judgeID, textChunk)
if err != nil {
LogPipelineError(runID, "ElevenLabs Chunk Generation Failed", err)
return
}
defer resp.Body.Close()
// Step 2: Stream chunked binary bytes directly to your object file storage destination
storageURL, _ := StreamBinaryToStorage(runID, judgeID, index, resp.Body)
// Step 3: Write partial segments directly into database rows
SaveAudioClipRow(runID, judgeID, index, storageURL)
// Step 4: Emit a WebSocket event to live-update the corresponding web UI player node
EmitWebSocketAudioReady(runID, judgeID, index, storageURL)
}(idx, chunk)
}
}
func SplitScriptIntoSentences(text string, maxSentences int) []string {
// Splits text logically by punctuation marks without corrupting context flows
sentences := strings.SplitAfter(text, ". ")
var chunks []string
var currentChunk []string
for _, sentence := range sentences {
currentChunk = append(currentChunk, sentence)
if len(currentChunk) &gt;= maxSentences {
chunks = append(chunks, strings.Join(currentChunk, " "))
currentChunk = nil
}
}
if len(currentChunk) &gt; 0 {
chunks = append(chunks, strings.Join(currentChunk, " "))
}
return chunks
}
